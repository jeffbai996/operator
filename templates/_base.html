(function () { const badge = document.getElementById("operator-badge"); if (!badge) return; const base = "{{ url_for('operator.operator_unseen') }}"; async function refresh() { try { const r = await fetch(base, { cache: "no-store" }); const d = await r.json(); const n = d.count || 0; if (n > 0) { badge.textContent = n > 99 ?
